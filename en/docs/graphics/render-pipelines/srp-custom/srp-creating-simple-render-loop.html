            // ä½¿ç”¨å‰”é™¤å‚æ•°æ‰§è¡Œå‰”é™¤æ“ä½œï¼Œå¹¶å­˜å‚¨ç»“æœ
            var cullingResults = context.Cull(ref cullingParameters);

            // åŸºäºå½“å‰æ‘„åƒæœºï¼Œæ›´æ–°å†…ç½®ç€è‰²å™¨å˜é‡çš„å€¼
            context.SetupCameraProperties(camera);

            // åŸºäº LightMode é€šé“æ ‡ç­¾å€¼ï¼Œå‘ Unity å‘ŠçŸ¥è¦ç»˜åˆ¶çš„å‡ ä½•ä½“
            ShaderTagId shaderTagId = new ShaderTagId("ExampleLightModeTag");

            // åŸºäºå½“å‰æ‘„åƒæœºï¼Œå‘ Unity å‘ŠçŸ¥å¦‚ä½•å¯¹å‡ ä½•ä½“è¿›è¡Œæ’åº
            var sortingSettings = new SortingSettings(camera);

            // åˆ›å»ºæè¿°è¦ç»˜åˆ¶çš„å‡ ä½•ä½“ä»¥åŠç»˜åˆ¶æ–¹å¼çš„ DrawingSettings ç»“æ„
            DrawingSettings drawingSettings = new DrawingSettings(shaderTagId, sortingSettings);

            // å‘ŠçŸ¥ Unity å¦‚ä½•è¿‡æ»¤å‰”é™¤ç»“æœï¼Œä»¥è¿›ä¸€æ­¥æŒ‡å®šè¦ç»˜åˆ¶çš„å‡ ä½•ä½“
            // ä½¿ç”¨ FilteringSettings.defaultValue å¯æŒ‡å®šä¸è¿›è¡Œè¿‡æ»¤
            FilteringSettings filteringSettings = FilteringSettings.defaultValue;
        
            // åŸºäºå®šä¹‰çš„è®¾ç½®ï¼Œè°ƒåº¦å‘½ä»¤ç»˜åˆ¶å‡ ä½•ä½“
            context.DrawRenderers(cullingResults, ref drawingSettings, ref filteringSettings);

            // åœ¨éœ€è¦æ—¶è°ƒåº¦å‘½ä»¤ç»˜åˆ¶å¤©ç©ºç›’
            if (camera.clearFlags == CameraClearFlags.Skybox && RenderSettings.skybox != null)
            {
                context.DrawSkybox(camera);
            }

            // æŒ‡ç¤ºå›¾å½¢ API æ‰§è¡Œæ‰€æœ‰è°ƒåº¦çš„å‘½ä»¤
            context.Submit();
        }
    }
